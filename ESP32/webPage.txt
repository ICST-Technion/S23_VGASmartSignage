            function toggleTemplateOptions() {
                var modeSelect = document.getElementById('mode-select').value;
                var templateOptions = document.getElementById('template-options');
                var templateInputs = document.getElementById('template-inputs');
    
                if (modeSelect === 'manual') {
                    templateOptions.style.display = 'block';
                    templateInputs.style.display = 'none';
                } else {
                    templateOptions.style.display = 'none';
                    templateInputs.style.display = 'none';
                }
            }
